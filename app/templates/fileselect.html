{# fileselect.html Template für die Auswahl von Files zum Upload. Parameter zur Ausgestaltung: {{ action }} : die Route, die nach der File-Auswahl aufgeufen wird {{ filetype }} : vorgesehener Dateityp = '.zip' oder leer, wenn Diretory-Auswahl {{ description }} Typ von vorgesehenen Dateien ('uploadroom', 'uploaddb', 'uploadcsv') #}
{% if description == "uploadroom" %} Raumdaten (zip) hochladen {% elif description == "uploadcsv" %} CSV-Datei hochladen {% elif description == "uploaddb" %} User-Datenbank hochladen {% endif %}

{% if description == "uploadroom" %} Die Daten der ausgewählten ZIP-Datei werden in den aktuellen Raum integriert. {% elif description == "uploadcsv" %} Die ausgewählte CSV-Datei wird in den Raum hochgeladen. Eine existierende CSV-Datei mit gleichem Namen wird gelöscht. {% elif description == "uploaddb" %} Die ausgewählte User-Datenbank ersetzt die aktuelle User-Datenbank. {% endif %}
